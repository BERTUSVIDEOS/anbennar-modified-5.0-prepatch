malus_if_dwarven_mechanic = {
	potential = {
		has_race_recruitment = {
			race = elven
		}
		has_race_recruitment = {
			race = dwarven
		}
	}
	trigger = {
		has_race_recruitment = {
			race = dwarven
		}
	}
	elven_loyalty_modifier = -0.1
	land_morale = -0.05
}

malus_if_monstrous_mechanic = {
	potential = {
		OR = {
			has_race_recruitment = {
				race = elven
			}
			has_race_recruitment = {
				race = dwarven
			}
			has_race_recruitment = {
				race = half_elven
			}
		}
		has_country_modifier = monstrous_nation
	}
	trigger = {
		has_country_modifier = monstrous_nation
	}
	dwarven_loyalty_modifier = -0.1
	elven_loyalty_modifier = -0.1
	half_elven_loyalty_modifier = -0.1
	discipline = -0.05
}

	potential = {
		OR = {
			has_race_recruitment = {

			}

			}

			}
			has_race_recruitment = {
				race = halfling
			}
			has_race_recruitment = {
				race = gnomish
			}
			has_race_recruitment = {
				race = orcish
			}
			has_race_recruitment = {
				race = goblin
			}
			has_race_recruitment = {
				race = kobold
			}
		}
	}
	movement_speed = 0.10
}
