estate_troll_recruitment = {
	icon = privilege_demand_more_troops
	loyalty = -0.10
	influence = 0.10
	is_valid = {
		NOT = {mil_tech = 2}
	}
	can_select = {
		high_tolerance_troll_race_trigger = yes
		any_owned_province = {
			has_troll_majority_trigger = yes
		}
	}
	on_granted = {
		hidden_effect = {
			set_country_flag = estate_troll_recruitment_flag
			country_event = { id = bertus_racial_military.1001 }
			if = {
				limit = { NOT = { has_country_flag = estate_troll_recruitment_cooldown_flag } }
				set_country_flag = estate_troll_recruitment_cooldown_flag
			}
		}
	}
	modifier_by_land_ownership = {
	}
	penalties = {
	}
	benefits = {
		has_samurai = yes
	}
	mechanics = {
		
	}
	conditional_modifier = {
		trigger = { estate_territory = { estate = estate_troll territory = 90 } }
		modifier = {
			allowed_samurai_fraction = 1.0
		}
	}
	can_revoke = {
		had_country_flag = {
			flag = estate_troll_recruitment_cooldown_flag
			days = 7300
		}
	}
	on_revoked = {
		hidden_effect = {
			clr_country_flag = estate_troll_recruitment_flag
			clr_country_flag = estate_troll_recruitment_cooldown_flag
			country_event = { id = bertus_racial_military.1001 }
		}
	}
	on_invalid = {
		clr_country_flag = estate_troll_recruitment_flag
		update_troll_recruitment = yes		
		set_estate_privilege = estate_troll_recruitment_2
	}
	ai_will_do = {
		factor = 1000
	}
}
