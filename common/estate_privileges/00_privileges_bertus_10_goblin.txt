	}

estate_goblin_recruitment = {
	icon = privilege_demand_more_troops
	loyalty = -0.10
	influence = 0.10

		NOT = {mil_tech = 2}

	}

		hidden_effect = {
			set_country_flag = estate_goblin_recruitment_flag
			country_event = { id = bertus_racial_military.1001 }

				limit = { NOT = { has_country_flag = estate_goblin_recruitment_cooldown_flag } }
				set_country_flag = estate_goblin_recruitment_cooldown_flag
			}
		}
	}
	modifier_by_land_ownership = {
		
	}
	penalties = {
	}

	}
	mechanics = {
		
	}
	can_revoke = {
		had_country_flag = {
			flag = estate_goblin_recruitment_cooldown_flag
			days = 7300
		}
	}
	on_revoked = {
		hidden_effect = {
			clr_country_flag = estate_goblin_recruitment_flag
			clr_country_flag = estate_goblin_recruitment_cooldown_flag
			country_event = { id = bertus_racial_military.1001 }
		}
	}
	on_invalid = {
		clr_country_flag = estate_goblin_recruitment_flag
		update_goblin_recruitment = yes
		set_estate_privilege = estate_goblin_recruitment_2

	}
