	icon = privilege_recruit_minister
	influence = 0.15
	
	can_select = {
		NOT = {	
			estate_influence = {
			estate = estate_nationalist
			influence = 75
			}
		}
	}
	
	on_granted = {
		if = {
				limit = {
					monthly_income = 25
				}
				define_advisor = {
					type = idea_var_advisor_19
					skill = 3
					cost_multiplier = 0.5
				}
			}
			if = {
				limit = { NOT = { monthly_income = 25 } }
				define_advisor = {
					type = idea_var_advisor_19
					skill = 2
					cost_multiplier = 0.5
				}
			}				
	}
	
	penalties = {
	}
	
	cooldown_years = 10
	on_cooldown_expires = {
		remove_estate_privilege = estate_priv_idea_variation_12_2
	}

	ai_will_do = {
		factor = 10
		modifier = {
			factor = 0
				estate_influence = {
					estate = estate_nationalist
					influence = 80
				}
		}
	}
}
