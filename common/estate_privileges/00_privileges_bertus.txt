
estate_artificers_emergency_general = {
	icon = privilege_promote_maratha_nobility
	loyalty = -0.40
	can_select = {
		NOT = {
			has_estate_privilege = estate_vampires_vampiric_military_command
		}
		artifice_has_points_for_masterwork = yes
		NOT = {has_government_attribute = magic_commander_academy}
	}
	on_granted = {
		add_estate_loyalty = {
			estate = estate_artificers
			loyalty = -20
		}
		if = {
			limit = {
				full_idea_group = generalstab0
			}
			get_artificers_general_ideas = yes
		}
		else = {
			get_artificers_general = yes
		}
		artifice_granted_masterwork = yes
	}
	penalties = {
		army_tradition_from_battle = -0.50
	}
	benefits = {
	}
	cooldown_years = 20
	on_revoked = {		
		artifice_remove_masterwork = yes
	}
	on_cooldown_expires = {
		remove_estate_privilege = estate_artificers_emergency_general
	}
	ai_will_do = {
		factor = 0
	}
}
