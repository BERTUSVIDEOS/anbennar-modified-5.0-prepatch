
estate_adventurers_emergency_general = {
	icon = privilege_promote_maratha_nobility
	can_select = {
		NOT = {has_estate_privilege = estate_vampires_vampiric_military_command}
		NOT = {has_estate_privilege = estate_mages_emergency_general}
		NOT = {has_estate_privilege = estate_artificers_emergency_general}
		NOT = {has_government_attribute = artificer_commander_academy}
		NOT = {has_government_attribute = magic_commander_academy}
		NOT = {has_government_attribute = adventurer_commander_academy}
	}
	on_granted = {
		add_estate_loyalty = {estate = estate_adventurers loyalty = -20}
		set_country_flag = get_adventurer_flag
		custom_tooltip = get_great_hero_general_tt
		hidden_effect = {
			set_saved_name = {key = special_general_name type = advisor scope = ROOT}
			country_event = {id = bertus_changes.6030}
		}
	}
	penalties = {
		army_tradition_from_battle = -0.50
	}
	benefits = {}
	cooldown_years = 20
	on_revoked = {}
	on_cooldown_expires = {
		remove_estate_privilege = estate_adventurers_emergency_general
	}
	ai_will_do = {factor = 0}
}

estate_adventurers_conquistador = {
	icon = privilege_promote_maratha_nobility
	influence = 0.1
	can_select = {
		estate_loyalty = {estate = estate_adventurers loyalty = 30}
	}
	on_granted = {
		add_mil_power = -50
		add_dip_power = -50
		if = {
			limit = {
				estate_influence = {
					estate = estate_adventurers
					influence = 100
				}
			}
			create_conquistador = { 
				tradition = 60 
				add_shock = 2
				add_fire = 2
				add_manuever = 2
			}
		}
		else_if = {
			limit = {
				estate_influence = {
					estate = estate_adventurers
					influence = 90
				}
			}
			create_conquistador = { 
				tradition = 60 
				add_shock = 1
				add_fire = 1
				add_manuever = 2
			}
		}
		else_if = {
			limit = {
				estate_influence = {
					estate = estate_adventurers
					influence = 80
				}
			}
			create_conquistador = { 
				tradition = 60 
				add_shock = 1
				add_fire = 1
				add_manuever = 1
			}
		}
		else_if = {
			limit = {
				estate_influence = {
					estate = estate_adventurers
					influence = 70
				}
			}
			create_conquistador = { 
				tradition = 80 
				add_shock = 1
				add_fire = 1
				add_manuever = 1
			}
		}
		else_if = {
			limit = {
				estate_influence = {
					estate = estate_adventurers
					influence = 60
				}
			}
			create_conquistador = { 
				tradition = 40 
				add_manuever = 1
			}
		}
		else_if = {
			limit = {
				estate_influence = {
					estate = estate_adventurers
					influence = 50
				}
			}
			create_conquistador = { 
				tradition = 20 
				add_manuever = 1
			}
		}
		else = {
			create_conquistador = { tradition = 40 }
		}
	}
	penalties = {}
	benefits = {}
	cooldown_years = 20
	on_cooldown_expires = {
		remove_estate_privilege = estate_adventurers_conquistador
	}
	ai_will_do = {factor = 0}
}
