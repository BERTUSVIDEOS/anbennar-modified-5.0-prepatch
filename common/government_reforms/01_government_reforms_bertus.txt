wine_empire = {
	icon = "merchant"
	allow_normal_conversion = yes
	valid_for_nation_designer = yes
	nation_designer_cost = 0
	potential = {
		OR = {
			culture_group = lencori
			primary_culture = rosanda
		}
		OR = {
			has_reform = wine_empire
			have_had_reform = wine_empire
			government_rank = 3
		}
	}
	trigger = {
		OR = {
			culture_group = lencori
			primary_culture = rosanda
		}
		any_owned_province = {
			trade_goods = wine
		}
	}
	fixed_rank = 3
	modifiers = {
		elven_influence_modifier = 0.1
		elven_loyalty_modifier = 0.1
		global_autonomy = -0.025
		governing_capacity = 100	
	}
	custom_attributes = {
		winelord_boost = yes
	}
	effect = {
		hidden_effect = {
			every_owned_province = {
				if = {
					limit = { trade_goods = wine }
					add_province_modifier = {
						name = wine_lord_wine
						duration = -1
						hidden = yes
					}
				}
			}
		}
	}
	removed_effect = {
		hidden_effect = {
			every_owned_province = {
				limit = { has_province_modifier = wine_lord_wine }
				remove_province_modifier = wine_lord_wine
			}
		}
	}
}
