	every_country = {
		limit = {ai = no}

		if = {
			limit = { NOT = { check_variable = { which = global_slave_cost_var value = 10.0 } } }
			set_variable = { which = global_slave_cost_var value = 10.0 }
		}
	}

			limit = { NOT = { check_variable = { which = global_slave_cost_var which = ROOT } } }
			change_variable = { which = global_slave_cost_var which = ROOT }
		}
	}
}

	save_event_target_as = country_elder_advisor_target
	if = {
		limit = {has_country_modifier = elven_administration}
		random_owned_province = {
			limit = {
				OR = {
					has_owner_accepted_culture = yes
					has_owner_culture = yes
				}
				culture_group = elven
			}
			save_event_target_as = province_event_target
		}
	}
	else_if = {
		limit = {has_country_modifier = dwarven_administration}
		random_owned_province = {
			limit = {
				OR = {
					has_owner_accepted_culture = yes
					has_owner_culture = yes
				}
				culture_group = dwarven
			}
			save_event_target_as = province_event_target
		}
	}
	else_if = {
		limit = {has_country_modifier = gnomish_administration}
		random_owned_province = {
			limit = {
				OR = {
					has_owner_accepted_culture = yes
					has_owner_culture = yes
				}
				culture_group = gnomish
			}
			save_event_target_as = province_event_target
		}
	}
	else_if = {
		limit = {
			any_ally = {
				OR = {
					has_country_modifier = elven_administration
					has_country_modifier = dwarven_administration
					has_country_modifier = gnomish_administration
				}
			}
		}
		random_province = {
			limit = {
				owner = {
					alliance_with = ROOT
					OR = {
						has_country_modifier = elven_administration
						has_country_modifier = dwarven_administration
						has_country_modifier = gnomish_administration
					}
				}
				has_owner_accepted_culture = yes
				OR = {
					culture_group = elven
					culture_group = dwarven
					culture_group = gnomish
				}
			}
			save_event_target_as = province_event_target
		}
	}
	else_if = {
		limit = {
			any_owned_province = {
				OR = {
					has_integrated_elven_pop_trigger = yes
					has_integrated_dwarven_pop_trigger = yes
					has_integrated_gnomish_pop_trigger = yes
				}
			}	
		}
		random_owned_province = {
			limit = {
				OR = {
					has_integrated_elven_pop_trigger = yes
					has_integrated_dwarven_pop_trigger = yes
					has_integrated_gnomish_pop_trigger = yes
				}
			}
			save_event_target_as = province_event_target
		}
	}	
	else_if = {
		limit = {has_estate_privilege = estate_raj_ministries_ceremonial_ruler_act}
		random_owned_province = {
			limit = {has_owner_accepted_culture = yes}
			save_event_target_as = province_event_target
		}
	}	
	else_if = {
		limit = {is_papal_controller = yes}
		random_province = {
			limit = {religion = ravelian has_owner_accepted_culture = yes}
			save_event_target_as = province_event_target
		}
	}
	else = {
		random_owned_province = {
			limit = { has_owner_accepted_culture = yes }
			save_event_target_as = province_event_target
		}
	}
}

get_manpower_fraction = {
	FROM = {
		set_variable = { which = current_manpower value = 0 }
		set_variable = { which = manpower_to_transfer value = 1 }
		subtract_variable = { which = manpower_to_transfer value = $fraction$ }
		export_to_variable = { which = current_manpower value = manpower }
		multiply_variable = { which = manpower_to_transfer which = current_manpower }
		while = {
			limit = { check_variable = { which = current_manpower which = manpower_to_transfer } }
			subtract_variable = { which = current_manpower value = 1 }
			add_manpower = -1
			ROOT = {add_manpower = 1}
		}
	}
}
