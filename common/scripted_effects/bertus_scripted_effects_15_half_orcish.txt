update_half_orcish_tolerance = {
	hidden_effect = {
		while = {	# If it ends up above the max it puts it at the highest point (never above 100)
			limit = { check_variable = {which = half_orcish_race_tolerance which = max_race_tolerance} }
			change_variable = {which = half_orcish_race_tolerance value = -1}
			add_prestige_or_monarch_power = { amount = 1 }
		}
		if = {	# If it ends up below 0 it adds back up to 1
			limit = {NOT = {check_variable = {which = half_orcish_race_tolerance value = 1}}}
			set_variable = {which = half_orcish_race_tolerance value = 1}
		}
		clr_country_flag = just_updated_half_orcish_tolerance_cd_flag
		set_country_flag = just_updated_half_orcish_tolerance_cd_flag
		country_event = {id = half_orcish_tolerance_events.5004} 	# half_orcish tolerance provinces update 
	}
}

add_half_orcish_minority_size_effect = {
	custom_tooltip = add_half_orcish_minority_size_effect_tooltip
	hidden_effect = {
		#Impact Oppressed
		if = {
			limit = {has_province_modifier = half_orcish_minority_oppressed_small}
			remove_province_modifier = half_orcish_minority_oppressed_small
			add_permanent_province_modifier = { name = half_orcish_minority_oppressed_large  duration = -1}
		}
		#Impact Coexisting
		else_if = {
			limit = {has_province_modifier = half_orcish_minority_coexisting_small}
			remove_province_modifier = half_orcish_minority_coexisting_small
			add_permanent_province_modifier = { name = half_orcish_minority_coexisting_large duration = -1}
		}
		#Impact Integrated
		else_if = {
			limit = {has_province_modifier = half_orcish_minority_integrated_small}
			remove_province_modifier = half_orcish_minority_integrated_small
			add_permanent_province_modifier = { name = half_orcish_minority_integrated_large duration = -1}
		}
		else_if = {
			limit = {
				OR = {
					has_half_orcish_majority_trigger = yes
					has_province_modifier = half_orcish_minority_oppressed_large
					has_province_modifier = half_orcish_minority_coexisting_large
					has_province_modifier = half_orcish_minority_integrated_large
				}
			}
			add_1_dev_to_provinces = yes
		}
		else_if = {
			limit = {has_half_orcish_minority_trigger = no}
			add_permanent_province_modifier = { name = half_orcish_minority_coexisting_small duration = -1}
		}
	}
}
