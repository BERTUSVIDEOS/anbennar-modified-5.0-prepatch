get_levies_from_dev = {
	hidden_effect = {
		set_variable = {
			which = total_development_var
			value = 0
		}
		set_variable = {
			which = cicles_per_development
			value = 1
		}
		every_owned_province = {
			limit = {
				culture_group = ROOT
				NOT = {has_province_modifier = recruited_levy_modifier}
			}
			add_province_modifier = {
				name = recruited_levy_modifier
				duration = 3650
			}
			set_variable = {
				which = total_development_var
				value = 0
			}
			export_to_variable = {
				which = total_development_var
				value = development
			}
			ROOT = {
				change_variable = {
					which = total_development_var
					which = PREV
				}
			}
		}
		while = {
			limit = {
				check_variable = {
					which = total_development_var
					which = cicles_per_development
				}
			}
			change_variable = {
				which = cicles_per_development
				value = $dev_per_levy$
			}
			capital_scope = {
				carolean_infantry  = ROOT
			}
		}
	}
