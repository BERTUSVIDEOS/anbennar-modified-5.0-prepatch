
ibevar_reformation = {
	potential = {
		tag = A32
		NOT = { has_country_flag = ibevar_reformation_end }
	}

	can_start = {
		OR = {
			NOT = {is_year = 1475}
			AND = {
				is_year = 1475
				religion = regent_court
			}
		}
	}
	
	can_stop = {
        always = no
	}
	
	progress = {
		modifier = {
			factor = 100
			always = yes
		}
	}
	
	can_end = {
		if = {
			limit = { ai = yes }
			OR = {
				AND = {
					south_ibevar_area = {
						type = all
						owned_by = ROOT
						religion = regent_court
					}
					north_ibevar_area = {
						type = all
						owned_by = ROOT
						religion = regent_court
					}
					ibevar_area = {
						type = all
						owned_by = ROOT
						religion = regent_court
					}
				}
				has_country_flag = ibevar_ai_bypass
			}
		}
		else = {
			OR = {
				had_country_flag = {
					flag = ibevar_stays_forebear
					days = 10950
				}
				AND = {
					south_ibevar_area = {
						type = all
						owned_by = ROOT
						religion = regent_court
					}
					north_ibevar_area = {
						type = all
						owned_by = ROOT
						religion = regent_court
					}
					ibevar_area = {
						type = all
						owned_by = ROOT
						religion = regent_court
					}
				}
			}
		}
	}
	
	modifier = {
		stability_cost_modifier = 0.5
		tolerance_heathen = -2
		tolerance_heretic = -2
		tolerance_own = -2
	}
	
	on_start = ibevar.2
	on_end = ibevar.7
	
	on_monthly = {
		events = {
			ibevar.102
		}
		random_events = { 
		}
	}
}
