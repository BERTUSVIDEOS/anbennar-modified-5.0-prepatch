	potential = {
		tag = A32
		NOT = { has_country_flag = ibevar_reformation_end }
	}

	can_start = {
		NOT = { has_country_modifier = ibevar_old_tradition_crisis_solved}
		NOT = {religious_unity = 0.9}
		is_part_of_hre = yes
	}
	
	can_stop = {
		always = no
	}
	
	progress = {
		modifier = {
			factor = 100
			always = yes
		}
	}
	
	can_end = {
		religious_unity = 0.9
		NOT = {
			any_owned_province = {
				OR = {
					has_province_modifier = ibevar_old_tradition
					has_province_modifier = ibevar_reformation_effort
					has_province_modifier = ibevar_last_reformation_effort
				}
			}
		}
	}
	
	modifier = {
		stability_cost_modifier = 0.5
		tolerance_heathen = -2
		tolerance_heretic = -2
		tolerance_own = -2
	}
	
	on_start = ibevar.2
	on_end = ibevar.7
	
	on_monthly = {
		events = {
			ibevar.102
		}
		random_events = { 
		}
	}
}
