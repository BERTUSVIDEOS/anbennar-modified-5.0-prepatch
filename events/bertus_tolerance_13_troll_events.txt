	trigger = {
		owner = {NOT = {has_country_modifier = troll_administration}}
		has_any_troll_pop_trigger = yes
	}
	mean_time_to_happen = {
		days = 73000
		modifier = {factor = 3.00 owner = {NOT = {had_country_flag = { flag = troll_tolerance_events_1001_flag days = 1825}}}}
		modifier = {factor = 2.0 owner = {any_owned_province = {culture_is_troll = yes ROOT = {employed_advisor = {culture = PREV}}}}}
		modifier = {factor = 0.75 owner = {has_country_modifier = racial_focus_troll}}
		modifier = {factor = 0.75 owner = {has_idea = tolerance_idea_2}}
		modifier = {factor = 0.90 owner = {medium_tolerance_troll_race_trigger = yes}}
		modifier = {factor = 5.0 has_small_troll_minority_trigger = yes}
		modifier = {factor = 3.0 has_large_troll_minority_trigger = yes}
		modifier = {factor = 0.90 development = 30}
		modifier = {factor = 0.90 development = 40}
	}
	immediate = {
		hidden_effect = {
			owner = {
				clr_country_flag = troll_tolerance_events_1001_flag 
				set_country_flag = troll_tolerance_events_1001_flag 
			}
			save_event_target_as = province_event_target
			if = {
				limit = {culture_is_troll = yes}
				save_event_target_as = advisor_province_event_target
			}
			else_if = {
				limit = {owner = {any_owned_province = {culture_is_troll = yes}}}
				owner = {
					random_owned_province = {
						limit = {culture_is_troll = yes}
						save_event_target_as = advisor_province_event_target
					}
				}
			}
			else = {
				random_province = {
					limit = {culture_is_troll = yes}
					save_event_target_as = advisor_province_event_target
				}
			}
			set_variable = {which = province_development value = 0}
			export_to_variable = {which = province_development value = development}
			set_variable = {which = cash_per_dev_tooltip value = 0}
			change_variable = {which = cash_per_dev_tooltip value = 2}
			set_variable = {which = prestige_per_dev_tooltip value = 0}
			change_variable = {which = prestige_per_dev_tooltip value = 0.2}
		}
	}

	# So what?
	option = {
		name = troll_tolerance_events.1001.a
		ai_chance = {factor = 50}
		
		add_unrest = 10
	}
	# Recruit
	option = {
		name = troll_tolerance_events.1001.b
		ai_chance = {factor = 50}
		owner = {
			get_racial_advisor = {
				race = troll
			}
