province_event = {
	id = sjr_lorent_set_up.1
	title = sjr_lorent_set_up.1.t
	desc = sjr_lorent_set_up.1.desc
	picture = DIPLOMACY_eventPicture
	
	hidden = yes

    trigger = {
		OR = {
			AND = { #cuando hay que ponerlo
				OR = {
					region = lencenor_region
					region = west_dameshead_region
					region = small_country_region
					area = beepeck_area
					area = iochand_area
				}
				NOT = {
					has_province_modifier = peace_under_lorent_modifier
				}
				owner = {
					OR = {
						tag = A01
						is_subject_of = A01
					}
					#NOT = { had_recent_war = 5 }
					#is_at_war = NO
				}
				owner = { is_at_war = NO }
				owner = { NOT = { had_recent_war = 2 } }
				NOT = { has_global_flag = peace_under_lorent_flag_off }
			}
			AND = {  #Cuando hay que quitarlo
				has_province_modifier = peace_under_lorent_modifier
				OR =  { 
					NOT = {
						owner = {
							OR = {
								tag = A01
								is_subject_of = A01
							}
						}
					}
					owner = { is_at_war = yes }
					has_global_flag = peace_under_lorent_flag_off
				}
			}
		}
    }
	
	immediate = {
	}
	option = {
		name = sjr_lorent_set_up.1.a
		trigger = {
			NOT = {
				has_province_modifier = peace_under_lorent_modifier
			}
		}
		add_province_modifier = { 
			name = peace_under_lorent_modifier
			duration = -1  
		}
	}
	option = {
		name = sjr_lorent_set_up.1.b
		trigger = {
			has_province_modifier = peace_under_lorent_modifier
		}
		remove_province_modifier = peace_under_lorent_modifier
	}
}

		NOT = {is_year = 1445}

		}

				name = lorent_player_modifier

				duration = -1
				desc = lorent_aristocratic_military_modifier_DESC_TOOLTIP
			}
			add_war_exhaustion = 10
