	trigger = {}
	hidden = yes
	
	is_triggered_only = yes
	
	immediate = {
		all_races_effect = { EFFECT = set_pop_percentage }
        change_estate_land_share = { estate = all share = -100}
		all_races_effect = { EFFECT = take_estate_land_share_all }
	}
	
	option = {
		name = "bertus_racial_military.1002.a"
		all_races_effect = { EFFECT = change_estate_land_to_percentage }
		
		give_estate_land_share_gigantic = { estate = all }
	}
	
	after = {
		while = {
			limit = { NOT = {crown_land_share = 0} }
			change_estate_land_share = { estate = all share = -0.1 }
		}
		if = {
			limit = {
				any_country = {
					war_with = ROOT
					any_province = {
						units_in_province = PREV
						OR = {
							units_in_province = ROOT
							any_neighbor_province = { 
								units_in_province = ROOT
							}
						}
					}
				}
			}
			country_event = { id = bertus_racial_military.1001 days = 30 }
		}
		else = {
			update_tech_group = yes
		}
		hidden_effect = { country_event = {id = bertus_changes.1042} }
	}
}
