	is_triggered_only = yes
    trigger = {	
    }
		
	immediate = {
	}
	option = {	#close
		name = bertus_artificers.50000.z
		highlight = yes
	}
	option = {	#
		name = bertus_artificers.1.a
		trigger = {			
			NOT = { has_estate_privilege = estate_artificers_organization_state }
			NOT = { has_estate_privilege = estate_artificers_organization_feiten_shuimu_cadre }
			NOT = { has_estate_privilege = artifice_invention_research_privilege }
			NOT = { has_estate_privilege = artifice_invention_directed_research }
			NOT = { has_estate_privilege = artifice_invention_directed_research_feiten_precursor_secrets }
			NOT = { has_country_flag = artificery_research_choice_ongoing }
		}
		custom_tooltip = set_research_privilege_tt
		hidden_effect = {
			set_estate_privilege = artifice_invention_research_privilege
			country_event = { id = artifice_events.1 }
		}
	}
	option = {	#
		name = bertus_artificers.1.b
		trigger = {
			NOT = { has_estate_privilege = artifice_invention_research_privilege }
			NOT = { has_estate_privilege = artifice_invention_directed_research }
			NOT = { has_estate_privilege = artifice_invention_directed_research_feiten_precursor_secrets }
			NOT = { has_country_flag = artificery_research_choice_ongoing }
			OR = { 
				has_estate_privilege = estate_artificers_organization_state
				has_estate_privilege = estate_artificers_organization_feiten_shuimu_cadre 
			}
		}
		if = { limit = { tag = event_target:feiten_or_was_feiten_target }	#Feiten
			if = {
				limit = {
					mission_completed = feiten_secrets_of_the_precursors
					OR = {
						any_subject_country = { any_owned_province = { trade_goods = precursor_relics } }
						any_owned_province = { trade_goods = precursor_relics }
					}
				}
				set_country_flag = artificery_research_choice_ongoing
				hidden_effect = { country_event = { id = artifice_events.10 } }
			}
			else = {
				custom_tooltip = set_research_privilege_tt
				hidden_effect = { set_estate_privilege = artifice_invention_directed_research }
			}
		}
		else = {
			custom_tooltip = set_research_privilege_tt
			hidden_effect = { set_estate_privilege = artifice_invention_directed_research }
		}
		hidden_effect = { country_event = { id = artifice_events.4 } }	#fast research
	}
	option = {	#Artificer Armory
		name = bertus_artificers.1.c
		country_event = {id = bertus_artificers.50000}
	}
	after = {
	}
}
