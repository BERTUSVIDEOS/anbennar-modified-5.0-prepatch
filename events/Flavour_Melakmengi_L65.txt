country_event = {
	id = flavor_melakmengi.2
	title = flavor_melakmengi.2.t
	desc = flavor_melakmengi.2.d
	picture = POPE_PREACHING_eventPicture
	
	trigger = {
		NOT = { exists = L65 }
		religion = mengi_myths
		culture_group = mengi
		NOT = { has_ruler_flag = refused_angel }
		is_at_war = no
		any_owned_province = {
			religion = mengi_myths
			OR = {
				region = mengi_region
				region = gebahiri_region
			}
		}
		capital_scope = {
			OR = {
				region = mengi_region
				region = gebahiri_region
			}
		}
	}
	
	mean_time_to_happen = {
		months = 3
		modifier = {
			factor = 0.5
			government = theocracy
			NOT = { has_adventurer_reform = yes }
		}
		modifier = {
			factor = 2.0
			capital_scope = {
				NOT = { religion = mengi_myths }
			}
		}
		modifier = {
			factor = 0.95
			theologian = 3
		}
		modifier = {
			factor = 0.75
			NOT = { num_of_cities = 3 }
		}
		modifier = {
			factor = 0.5
			NOT = { num_of_cities = 2 }
		}
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					religion = mengi_myths
					OR = {
						region = mengi_region
						region = gebahiri_region
					}
				}
				set_province_flag = papal_province
			}
			random_owned_province = {
				limit = {
					is_capital = no
					religion = mengi_myths
					OR = {
						region = mengi_region
						region = gebahiri_region
					}
				}
				set_province_flag = papal_province
			}
			random_owned_province = {
				limit = {
					is_capital = no
					religion = mengi_myths
					OR = {
						region = mengi_region
						region = gebahiri_region
					}
					owner = {
						NOT = {
							any_owned_province = {
								has_province_flag = papal_province
							}
						}
					}
				}
				set_province_flag = papal_province
			}
		}
	}
	
	option = {
		name = "flavor_melakmengi.2.a"
		ai_chance = { factor = 100 }
		random_owned_province = {
			limit = {
				has_province_flag = papal_province
			}
			remove_core = ROOT
			add_core = L65
			cede_province = L65
			clr_province_flag = papal_province
		}
		add_prestige = 25
		L65 = {
			change_religion = mengi_myths
		}

		add_prestige = 25
	}
	option = {
		name = "flavor_melakmengi.2.b"
		ai_chance = { factor = 0 }
		set_ruler_flag = refused_angel
		add_prestige = -25
	}
}
