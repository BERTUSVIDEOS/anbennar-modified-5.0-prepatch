country_event = {	#start peasants
	id = bertus_gawed_initial_disaster.1100
	title = bertus_gawed_initial_disaster.1100.t
	desc = bertus_gawed_initial_disaster.1100.desc
	picture = INVENTION_eventPicture
	
	is_triggered_only = yes
		
	immediate = {
		hidden_effect = {
			set_variable = { which = peasants_apeased_var value = 0 }
			
			set_variable = { which = temp value = 7 }
			while = {
				limit = { check_variable = { which = temp value = 1 } }
				change_variable = { which = temp value = -1 }
				random_owned_province = {
					limit = {
						culture_group = ROOT
						NOT = { development = 13}
						NOT = { 
							trade_goods = cloth
							trade_goods = carpet
							trade_goods = wool
						}
						NOT = {
							has_province_modifier = gawed_disaster_mages_control
							has_province_modifier = gawed_disaster_burghers_control
							has_province_modifier = gawed_disaster_nobles_control
							has_province_modifier = gawed_disaster_peasants_control
						}
					}
					add_province_modifier = { name = gawed_disaster_peasants_control duration = -1 }
				}
			}
			set_variable = { which = temp value = 4 }
			while = {
				limit = { check_variable = { which = temp value = 1 } }
				change_variable = { which = temp value = -1 }
				while = {
					limit = { check_variable = { which = temp value = 1 } }
					change_variable = { which = temp value = -1 }
					random_owned_province = {
						limit = {
							culture_group = ROOT
							has_province_modifier = gawed_disaster_peasants_control
						}
						save_event_target_as = province_target
						random_list = {
							30 = {
								spawn_rebels = { size = 1 type = leadered_peasant_rebels }
							}
							70 = {
								spawn_rebels = { size = 0.5 type = leadered_peasant_rebels }
							}
						}
					}
				}
			}
		}
		tooltip = {
			custom_tooltip = "Up to 4 provinces get:"
			event_target:province_target = {
				random_list = {
					30 = {
						spawn_rebels = { size = 1 type = leadered_peasant_rebels }
					}
					70 = {
						spawn_rebels = { size = 0.5 type = leadered_peasant_rebels }
					}
				}
			}
		}
	}
	option = {	#What a mess
		name = bertus_gawed_initial_disaster.1100.a
		custom_tooltip = "Peasants have taken control of up to 7 provinces"
	}
	after = {
	}
}
