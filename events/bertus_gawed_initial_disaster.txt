country_event = {	# bandits
	id = bertus_gawed_initial_disaster.1102
	title = bertus_gawed_initial_disaster.1102.t
	desc = bertus_gawed_initial_disaster.1102.desc
	picture = INVENTION_eventPicture
			
	is_triggered_only = yes
	mean_time_to_happen = {
		days = 1
		modifier = {
			factor = 2
			any_owned_province = { devastation = 10 }
		}
		modifier = {
			factor = 2
			any_owned_province = { devastation = 20 }
		}
	}
	trigger = {
		any_owned_province = {
			NOT = {
				has_terrain = city_terrain  #bandits cannot spawn in cities
				has_province_modifier = aw_bandits_1
				has_province_modifier = aw_bandits_2
				has_province_modifier = aw_bandits_3
			}
		}
	}
	immediate = {
		random_list = {
			20 = {
				trigger = {
					any_owned_province = {
						OR = {
							has_terrain = forest
							has_terrain = woods
							has_terrain = hills
						}
					}
				}
				random_list = {
					20 = {
						trigger = {
							any_owned_province = { devastation = 10 }
						}
						random_owned_province = {
							limit = {
								OR = {
									has_terrain = forest
									has_terrain = woods
									has_terrain = hills
								}
								devastation = 10
							}
							save_event_target_as = bandits_target
						}
					}
					10 = {
						random_owned_province = {
							limit = {
								OR = {
									has_terrain = forest
									has_terrain = woods
									has_terrain = hills
								}
							}
							save_event_target_as = bandits_target
						}
					}
				}
			}
			10 = {
				random_list = {
					20 = {
						trigger = {
							any_owned_province = { devastation = 10 }
						}
						random_owned_province = {
							limit = {
								devastation = 10
							}
							save_event_target_as = bandits_target
						}
					}
					10 = {
						random_owned_province = {
							limit = {
							}
							save_event_target_as = bandits_target
						}
					}
				}
			}
		}
	}
	option = {	#send troops
		name = bertus_gawed_initial_disaster.1102.a
		add_manpower = -2
		change_variable = { which = peasants_apeased_var value = 2 }
	}
	option = {	#cant afford
		name = bertus_gawed_initial_disaster.1102.b
		custom_tooltip = "Peasants wont like that"
		change_variable = { which = peasants_apeased_var value = -1 }

		hidden_effect = {
			event_target:bandits_target = {
				province_event = { id = aw_bandits.0 }				
			}
		}
	}
	after = {
	}
}
